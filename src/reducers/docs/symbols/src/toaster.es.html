1 import {
  2   SHOW_SNACKBAR,
  3   HIDE_SNACKBAR
  4 } from '../constants/ToasterActionTypes'
  5 
  6 // Set the initial state for the snackbar
  7 const initialState = {
  8   isSnackbarActive: false,
  9   snackbarText: "",
 10   snackbarTimeout: 2750
 11 }
 12 
 13 /**
 14  * Update the snackbar section of the store
 15  * @param  {object} state  current state
 16  * @param  {object} action action to apply
 17  * @return {object}        new state
 18  */
 19 function snackbar(state, action) {
 20   const {payload} = action
 21   const {snackbarText} = payload
 22   const newSnackbar = {...state, snackbarText, isSnackbarActive: true}
 23 
 24   return {
 25     ...newSnackbar
 26   }
 27 }
 28 
 29 /**
 30  * The toaster reducer
 31  * @param  {object} [state=initialState] state of the reducer
 32  * @param  {object} action               action to apply
 33  * @return {object}                      updated state
 34  */
 35 export const toaster = (state = initialState, action) => {
 36   switch (action.type) {
 37     case SHOW_SNACKBAR:
 38       return snackbar(state, action)
 39     case HIDE_SNACKBAR:
 40       return initialState
 41     default:
 42       return state
 43   }
 44 }
